UI-Notify

The UI-Notify is a simple message alert


                                   
<!--Here to show the notification-->
<div class="col-md-12" id="notify"></div>
<!-- Call Success notificaction-->
<button type="button" class="btn btn-default" onclick="UINotify.success('Success notification!');">Success</button>
<!-- Call Info notificaction -->
<button type="button" class="btn btn-default" onclick=" UINotify.info('Info notification!');">Info</button>
<!-- Call Warning notificaction -->
<button type="button" class="btn btn-default" onclick="UINotify.warning('Warning notification!');">Warning</button>
<!-- Call Danger notificaction-->
<button type="button" class="btn btn-default" onclick="UINotify.danger('Danger notification!');">Danger</button>
<!-- Call Wait notificaction -->
<button type="button" class="btn btn-default" onclick="UINotify.wait('Wait notification!');">Wait...</button>
<!-- Call Close method -->
<button type="button" class="btn btn-default" onclick="UINotify.close();">Close</button>
                           
                      
                                   
/*
* Create Object UI-Notify with element html parameter
*/
var UINotify = new UI.Notify(document.getElementById('notify'));

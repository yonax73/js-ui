UI-Notify

The UI-Notify is a simple message alert

Example


                            
<!-- Element for show the notification -->
<div class="col-md-12" id="notify"></div>
                           
                      
                        
/*
* Create new instance for UINotify with parameter html element id
*/
var UINotify = new UI.Notify('notify');
//Notification Success
UINotify.success('This is an example of success notification.');
//Method for close notification
UINotify.close();

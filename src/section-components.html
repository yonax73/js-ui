UI-Notify

Simple message alert

Example

View more. 
                                    
<!--Here to show the notification-->
<div class="col-md-3" id="notify"></div>
                                     
                                    
                                    
/*
* Create Object UI-Notify
* @param HtmlElement
*/
var UINotify = new UI.Notify(document.getElementById('notify'));
/*
* Show notification type Success
* @param String
*/
UINotify.success('Success Notify');

* Create Object UI-DropDownList
* @param HtmlElement
* @param Items []
*/
var UIDropDownList = new UI.DropDownList(document.getElementById('dropdownlist'), [
    { option: 1, value: 'Product A' },
    { option: 2, value: 'Product B' },
    { option: 3, value: 'Product C' },
    { option: 4, value: 'Product D' },
    { option: 5, value: 'Product E' },
    { option: 6, value: 'Product F' }
]);
/**
* Initialize the UI-DropDownList
* @param option
*/
UIDropDownList.init(5);

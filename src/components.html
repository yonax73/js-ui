Notify Object

The Notify Object is used for show messages

View api.    |    Try it yourself.

var notify = new UI.Notify(document.getElementById('notify'));

View api.    |    Try it yourself.

<div class="col-sm-2">
   <input id="input" class="form-control" placeholder="option"/>

 var products = new UI.DropDownList(document.getElementById('dropdownlist'), [

 ]);      
 
 document.getElementById('btn').onclick = function(){
     var option = document.getElementById('input').value;
     if(!option.isEmpty()){
       var item = products.getItemByOption(option);
       document.getElementById('option').textContent = item.option;
       document.getElementById('value').textContent = item.value;
       console.log(item);              
     }
 };

addItem(option:Any,value:Any)

View api.    |    Try it yourself. 
                                              
<!--Here to show the dropdownlist-->
<div class="col-sm-4" id="dropdownlist"></div>
<div class="col-sm-2">
   <input id="input-option" class="form-control" placeholder="option"/>                            
</div>
<div class="col-sm-2">
  <input id="input-value" class="form-control" placeholder="value"/>   
</div>
<div class="col-sm-2">
   <button id="btn" class="btn btn-default">Add Item</button>    
</div> 
                                          
                                        
                                              
 var products = new UI.DropDownList(document.getElementById('dropdownlist'), [
    { option: 1, value: 'Product A' },
    { option: 2, value: 'Product B',selected:true },
    { option: 3, value: 'Product C' }
 ]);      
 
 document.getElementById('btn').onclick = function(){
     var option = document.getElementById('input-option').value;
     var value = document.getElementById('input-value').value;
     if(!option.isEmpty() && !value.isEmpty()){
         products.addItem(option,value);                
                  
     }
 };

* Use for edit border color
*/

   border-radius: 2px;    
}

/**
* Use for edit color and background style.
*/
.dropdownlist.background span, .dropdownlist.background input[type="text"][readonly], .dropdownlist-list li  {  
  background-color:  #444444;    
  color:  #000000;    
}
/**
* Use for edit color and background style when is disabled.
*/
.dropdownlist.disabled span, .dropdownlist.disabled input[type="text"]  { 
  background-color:  #767676;    
  color:  #cccccc;    
  cursor:  not-allowed
}
/**
* Use for edit color and background style when is hover and selected.
*/
.dropdownlist-list  li:hover, .dropdownlist-list .selected  {    
  background-color:  #ffffff;    
  color:  #ff0000    
}

Notify Object

The Notify Object is used for show messages

success(message:String)

View api.    |    Try it yourself. 
                                            
<!--Here to show the notification-->
<div class="col-md-3" id="notify"></div>
<button  class="btn btn-success" id="btnSuccess" >Show Notification</button>
                                         
                                        
                                            
var notify = new UI.Notify(document.getElementById('notify'));
document.getElementById('btnSuccess').onclick = function () {          
    notify.success('Success Notify');
}

UI-Dropdownlist

Simple drop down list

Example

View more. 
                                        
<!--Here to show the dropdownlist-->
<div class="col-sm-4" id="dropdownlist"></div>
<div class="col-sm-2">
   <button id="btn" class="btn btn-default">Get Option</button>    
</div>
<div class="col-sm-2">
   <h5 >Output:  <b  id="output"></b> </h5>
</div>
                                     
                                    
                                        
 var products = new UI.DropDownList(document.getElementById('dropdownlist'), [
    { option: 1, value: 'Product A' },
    { option: 2, value: 'Product B' },
    { option: 3, value: 'Product C' }
 ]);
 
 products.init(2);
 
 document.getElementById('btn').onclick = function(){
     var option = products.getOption();           
     document.getElementById('output').textContent = option;
     console.log(option);
 };
